\begin{frame}
\frametitle{Projeto de controladores baseados em dados}
\framesubtitle{Critérios de performance}

\begin{itemize}
  \item Seguimento de referência,
  \item Rejeição ao ruído e 
  \item Uso reduzido de esforço de controle. 
\end{itemize}

Para o objetivo de \textcolor{blue}{seguimento de referência}, a performance pode ser avaliada pela norma:
\begin{equation}
J_y(\theta)\overset{\underset{\mathrm{\Delta}}{\,}}{=}  \bar{E} \left [ y(t)-y_d(t) \right ]^2 = \bar{E}\left [
(T(z,\theta)-T_d(z))r(t) \right ]^2
\nonumber
\end{equation}

\begin{block}{Controlador ideal - Seguimento de referência}
\begin{equation}
C_d(z)=\frac{T_d(z)}{G_0(z)(1-T_d(z))}
\nonumber
\end{equation}
\end{block}
\end{frame}

\begin{itemize}
  \item O método VRFT faz com que a função custo a ser minimizada seja quadrática em $\theta$, não recaindo em mínimos
  locais.
\end{itemize}
\begin{equation}
J_{VR}^N(\theta)=\frac{1}{N}\sum_{t=1}^{N}(u_L(t)-\varphi_L^T(t)\theta)^2
\nonumber
\end{equation}

\begin{equation}
\varphi_L(t)=\beta(z)\epsilon_L(t)
\nonumber
\end{equation}
