\section{Conclusões} 
%-------------------------------------------------------------------------------------------------

\begin{frame}
\frametitle{Conclusões}

\begin{itemize}
	\item Utilização de algoritmos de identificação de sistemas não lineares pode ser extendida para a identificação de
	controladores não lineares.
	\item Referência virtual traz diversas vantagens para a obtenção dos dados para alimentação dos algoritmos não
	lineares.
	\item Resultados obtidos com os exemplos apresentados podem ser consiederados satisfatórios para um grande número de
	aplicações.
\end{itemize}
\end{frame}
