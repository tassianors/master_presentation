\section{Definições} 
%-------------------------------------------------------------------------------------------------
\begin{frame}
\frametitle{Definições}

\begin{block}{Foco em sistemas SISO}
Sistemas {\it{Single input single output}} (SISO) discretos:
\begin{equation}
y(t)=G_0(z)u(t)+H_0(z)e(t)
\nonumber
\end{equation}
onde $G_0(z)$ é a representação da planta real do sistema, $H_0(z)$ é a representação do filtro que atua sobre o ruído
branco $e(t)$. $y(t)$ é a saída e $u(t)$ é a entrada do sistema.

\end{block}
\end{frame}

%-------------------------------------------------------------------------------------------------
\subsection{Identificação de sistemas} 
%-------------------------------------------------------------------------------------------------

\begin{frame}
\frametitle{Definições}
\framesubtitle{Identificação de sistemas}
Identificação de sistemas contém três componentes principais:

\begin{itemize}
	\item O sistema real a ser identificado $\mathcal{S}$
 	\begin{equation}
		\mathcal{S}:\;\; y(t)=G_0(z)u(t)+H_0(z)e(t)
		\label{eq:si_intro_true_system}
	\end{equation}

	\item A classe de modelos a ser utilizada na identificação $\mathcal{M}$
	\begin{equation}
		\mathcal{M}: \;\;\left \{ G(z, \theta), H(z, \theta) | \theta \in D_{\mathcal{M}} \right \}
		\label{eq:si_intro_model}
	\end{equation}

\end{itemize}
\end{frame}
%-------------------------------------------------------------------------------------------------
%-------------------------------------------------------------------------------------------------

\begin{frame}
\frametitle{Definições}
\framesubtitle{Identificação de sistemas}

\begin{itemize}
	\item Algum critério para elencar qual modelo dentro da classe de modelos melhor
	consegue representar o sistema $\mathcal{S}$ nas propriedades escolhidas.
	\begin{block}{Neste trabalho optou-se por:}
	\begin{equation}
		V(\theta)=\frac{1}{N}\sum_{t=1}^{N}\frac{1}{2} \varepsilon ^2(t, \theta)
		\label{eq:si_obj_etim_lsm_v}
	\end{equation}
	\end{block}
	onde $\varepsilon (t,\theta)$ é o erro de predição e pode ser definido como:
	
	\begin{equation}
		\varepsilon (t,\theta)=H^{-1}(z, \theta)\left \{ y(t)-G(z, \theta)u(t) \right \}
	\nonumber
	\end{equation}
\end{itemize}
\end{frame}

\subsection{Propriedades estatísticas das estimativas} 
%-------------------------------------------------------------------------------------------------
\begin{frame}
\frametitle{Definições}
\framesubtitle{Propriedades estatísticas das estimativas}

\begin{itemize}
  \item $\theta_0$: conjunto de parâmetros que faz com que:
  
  \begin{equation}
  	\left\{\begin{matrix}
G(z, \theta_0)\triangleq G_0(z)\\ 
H(z, \theta_0)\triangleq H_0(z)
\end{matrix}\right.
  \nonumber
  \end{equation}
  
  \item $\hat{\theta}_N$: Estimativa para um certo valor de $N$ pontos.
  \item $\theta^*$: Valor de convergência da estimativa quando $N \to \infty$:
  \begin{equation}
  \lim_{N \rightarrow \infty }\hat{\theta}_N = \theta^*
  \nonumber
  \end{equation}
\end{itemize}
\end{frame}
%-------------------------------------------------------------------------------------------------
%-------------------------------------------------------------------------------------------------

\begin{frame}
\frametitle{Definições}
\framesubtitle{Propriedades estatísticas das estimativas - Erros envolvidos}

\begin{block}{Erros atrelados às estimativas}
\begin{align*}
\text{Erro de variância} &= G(z, \hat{\theta}_N)-G(z, \theta^*)\\ 
\text{Erro de polarização}  &= G(z, \theta^*)-G_0(z) 
\end{align*}
\end{block}

Onde observa-se que o erro de polarização é a diferença entre o valor real $G_0(z)$ e a melhor aproximação possível
(quando $N\to\infty$) $G(z, \theta^*)$.
\vspace{\baselineskip}

Erro de variância é a diferença entre cada uma das estimativas obtidas e a estimativa quando $N \to \infty$ $G(z,
\theta^*)$
\end{frame}

\begin{frame}
\frametitle{Definições}
\framesubtitle{Propriedades estatísticas das estimativas - Elipse de confiança}

\begin{columns}
\column{0.5\textwidth}
\fontsize{8}{8}\selectfont
Região de confiança é definido pela elipse:

\begin{equation}
\tiny U_\theta=\left \{ \theta \mid (\hat{\theta}_N-\theta_0)^T P^{-1}_{N}(\hat{\theta}_N-\theta_0) \le
\chi^2_{\alpha}(n) \right \}
\nonumber
\end{equation}
onde:

\begin{equation}
\small P(\theta^*)\overset{\underset{\mathrm{\Delta}}{\,}}{=}\lim_{N\to
\infty}N\;E(\hat{\theta}_N-\theta^*)(\hat{\theta}_N-\theta^*)^T
\nonumber
\end{equation}
