\section{Objetivo e Motivação}
%-------------------------------------------------------------------------------------------------
\begin{frame}
\frametitle{Objetivo / Motivação}

\begin{block}{Objetivo}
Apresentar um estudo sobre projeto de controladores não lineares utilizando identificação de
sistemas e o conceito de referência virtual.
\end{block}

\begin{block}{Motivação}
\begin{itemize}
	\item Controladores não lineares são pouco explorados
	\item Para alguns tipos de sistemas, controladores não lineares conseguem ser mais performáticos
\end{itemize}
\end{block}

\end{frame}
